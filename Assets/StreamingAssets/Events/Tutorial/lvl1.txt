WHEN time:1 //spawn bob, tim, and sue each 5 seconds apart, then have them wander
    (DO player setgoal 3000000 )
    (DO npc.normal.bob spawn )
    (DO npc.normal.bob goto node.node14 )
    wait.5
    (DO npc.normal.david spawn )
    wait.5
    (DO npc.normal.sue spawn )
    wait.5
    (DO npc.normal.david wander )
    wait.5
    (DO npc.normal.sue wander )
    wait.5
    ;

WHEN time:15 //spawn 2 tentacles, then spawn tim and have him go to the kitchen
    (DO monster.tentacle.1 spawn room.dinningroom 15 )
    (DO monster.tentacle.1 spawn room.kitchen 15 )
    wait.15
    (DO npc.normal.tim spawn )
    wait.10
    (DO npc.normal.tim goto room.bathroom )
    ;

WHEN monster.tentacle.1:die if.flag.monsters=0
    wait.15
    (DO monster.tentacle.1 spawn room|dinningroom|kitchen|bathroom 5 )
    flag.monsters=1
    ;

WHEN monster.tentacle.1:die if.flag.monsters=1
    wait.15
    (DO monster.tentacle.1 spawn room|dinningroom|kitchen|bathroom 15 )
    (DO npc.normal.bob spawn )
    flag.monsters=2
    ;

WHEN monster.tentacle.1:die if.flag.monsters=2
    wait.15
    (DO monster.tentacle.1 spawn room|dinningroom|kitchen|bathroom 15 )
    flag.monsters=1
    ;

WHEN npc.normal.bob:exit
    wait.5
    (DO npc.normal.bob spawn )
    ;
